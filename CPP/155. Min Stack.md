# [155. Min Stack](https://leetcode.com/problems/min-stack/)
**千万注意：很容易自己打印的时候出错然后造成程序崩了，实际程序并没有错**
**比如我这里，在pop的时候，栈空的时候，printf s.top()就飞了**

```C++
class MinStack {
public:
    /** initialize your data structure here. */
    MinStack() {
    }
    
    void push(int x) {
        s.push(x);
        if(min_s.empty()) min_s.push(x);
        else if(x <= min_s.top()){
            min_s.push(x);
        }
    }
    
    void pop() {
        //printf("pop() before: s.top =%d, min_s.top=%d\n", s.top(), min_s.top());
        if(s.top() == min_s.top()){
            min_s.pop();
        }
        s.pop();
        //printf("pop() after: s.top =%d, min_s.top=%d\n", s.top(), min_s.top());
    }
    
    int top() {
        //printf("top: s.top = %d\n", s.top());
        return s.top();
    }
    
    int getMin() {
        //printf("get min: min_top = %d\n", min_s.top());
        return min_s.top();
    }
    
private:
    stack<int> s;
    stack<int> min_s;
};

/**
 * Your MinStack object will be instantiated and called as such:
 * MinStack obj = new MinStack();
 * obj.push(x);
 * obj.pop();
 * int param_3 = obj.top();
 * int param_4 = obj.getMin();
 */
```
