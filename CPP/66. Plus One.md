# [66. Plus One](https://leetcode.com/problems/plus-one/#/solutions)
* 自己的方法，就是trivial的

```C++
    vector<int> plusOne(vector<int>& digits) {
        vector<int> res;
        int len = digits.size();
        int carry = 1;
        for(int i = len-1; i>=0; i--){
            int tmp = digits[i];
            digits[i] = (digits[i]+carry) %10;
            carry = (tmp + carry) / 10;
        }
        
        if(carry == 0)
            res = digits;
        else{
            res.push_back(1);
            for(int i = 0; i<len; i++)
                res.push_back(digits[i]);
        }
        
        return res;
    }
```

* 直接模拟了手算加法的过程

```C++
    vector<int> plusOne(vector<int>& digits) {
        vector<int> res = digits;
        int n = digits.size();
        
        for(int i = n-1; i>=0; i--){
            if(res[i] == 9){
                res[i] = 0;
            }else{
                res[i]++;
                return res;
            }
        }
        
        res[0] = 1;
        res.push_back(0);
        return res;
    }
};
```
