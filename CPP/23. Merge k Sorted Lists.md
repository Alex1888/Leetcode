# [23. Merge k Sorted Lists](https://leetcode.com/problems/merge-k-sorted-lists/?tab=Solutions)

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
 // 我的方法：两两进行insert merge, 比较慢，beat 33%
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int len = lists.size();
        if(len == 0) return NULL;
        if(len == 1) return lists[0];
        ListNode* res = NULL;
        int start = 0;
        // 找到第一个非null
        for(int i = 0; i<len; i++){
            if(lists[i] != NULL){
                res = lists[i];
                start = i;
                break;
            }
        }
        
        // 两两进行insert merge
        for(int i=start+1; i<len; i++){
            if(lists[i] == NULL) continue;
            if(res->val < lists[i]->val)
                res = insertsort(res, lists[i]);
            else
                res = insertsort(lists[i], res);
        }
        
        return res;
    }
    
public:
// l1.value < l2.value
    ListNode* insertsort(ListNode* l1, ListNode* l2){
        ListNode* p1 = l1;
        ListNode* p2 = l2;
        while(p1 != NULL && p2 != NULL){
            // 找到第一个插入点
            while(p1->next != NULL && p1->next->val <= p2->val){
                p1=p1->next;
            }
            
            ListNode* tmp1 = p1->next;
            p1->next = p2;
            
            // 找到l2中插入的末尾节点
            while(p2->next != NULL && tmp1 != NULL && p2->next->val < tmp1->val){
                p2=p2->next;
                p1=p1->next;
            }
                
            ListNode* tmp2 = p2->next;
            p2->next = tmp1;
            p2 = tmp2;
            p1 = p1->next;
        }
        
        return l1;
    }
};

// 在这里看到两个解法：https://discuss.leetcode.com/topic/6882/sharing-my-straightforward-c-solution-without-data-structure-other-than-vector
// 这个解法非常简洁明了，可读性相当好，但是实际的时间还不如我写的
ListNode *mergeKLists(vector<ListNode *> &lists) {
    if(lists.empty()){
        return nullptr;
    }
    while(lists.size() > 1){
        lists.push_back(mergeTwoLists(lists[0], lists[1]));
        lists.erase(lists.begin());
        lists.erase(lists.begin());
    }
    return lists.front();
}
ListNode *mergeTwoLists(ListNode *l1, ListNode *l2) {
    if(l1 == nullptr){
        return l2;
    }
    if(l2 == nullptr){
        return l1;
    }
    if(l1->val <= l2->val){
        l1->next = mergeTwoLists(l1->next, l2);
        return l1;
    }
    else{
        l2->next = mergeTwoLists(l1, l2->next);
        return l2;
    }
}

// 实际思路和上面的是一样的
class Solution {
public:
    ListNode *mergeTwoLists(ListNode* l1, ListNode* l2) {
        if (NULL == l1) return l2;
        else if (NULL == l2) return l1;
        if (l1->val <= l2->val) {
            l1->next = mergeTwoLists(l1->next, l2);
            return l1;
        }
        else {
            l2->next = mergeTwoLists(l1, l2->next);
            return l2;
        }
    }
    ListNode *mergeKLists(vector<ListNode *> &lists) {
        if (lists.empty()) return NULL;
        int len = lists.size();
        while (len > 1) {
            for (int i = 0; i < len / 2; ++i) {
                lists[i] = mergeTwoLists(lists[i], lists[len - 1 - i]);
            }
            len = (len + 1) / 2;
        }
        
        return lists.front();
    }
};
```
