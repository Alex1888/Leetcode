# [224. Basic Calculator](https://leetcode.com/problems/basic-calculator/)
* 用栈保存当前计算的结果和加减号.
* 遇到数字时先计算当前的结果,遇到左括号入栈,然后重置res, sign
* 遇到右括号连续出栈,先出的是符号,再出的是上一次的结果

```c++
class Solution {
public:
    int calculate(string s) {
        stack<int> nums;
        int res =0;
        int sign = 1;
        int num = 0;
        
        for(int i=0; i<s.size(); i++){
            char c  = s[i];
            if(c >= '0'){
                int num = 0;
                while(i < s.size() && s[i] >= '0') {
                    num = 10 * num + s[i++] - '0';
                }
                res += sign*num;
                i--;
            }else if(c == '+'){
                sign = 1;
            }else if(c == '-'){
                sign = -1;
            }else if(c == '('){
                nums.push(res);
                nums.push(sign);
                res = 0;
                sign = 1;
            }else if(c == ')'){
                res *= nums.top(); nums.pop();
                res += nums.top(); nums.pop();
            }
        }
        
        return res;
    }
};

```
