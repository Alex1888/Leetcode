# [102. Binary Tree Level Order Traversal](https://leetcode.com/problems/binary-tree-level-order-traversal/)
*  第一版，不是很简洁

```C++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if (root == NULL) return res;
        vector<int> r{root->val};
        res.push_back(r);
        dfs(root, res, 1);
        
        return res;
    }
    
public:
    void dfs(TreeNode* root, vector<vector<int>>& res, int level){
        if(root != NULL){
            if (root->left == NULL && root->right == NULL) return;
            if (res.size() < level+1) {
                vector<int> temp;
                res.push_back(temp);
            }
            if(root->left != NULL) res[level].push_back(root->left->val);
            if(root->right!= NULL) res[level].push_back(root->right->val);
            if(root->left != NULL) dfs(root->left, res, level+1);
            if(root->right!= NULL) dfs(root->right, res, level+1);
        }
    }
```

*  第二版，标准dfs，而且注意用res.size()判断什么时候结束

```C++
class Solution {
public:
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        dfs(root, res, 0);
        return res;
    }
public:
    void dfs(TreeNode* root, vector<vector<int>>& res, int depth){
        if(root == NULL) return;
        if(res.size() == depth){
            res.push_back(vector<int>());
        }
        
        res[depth].push_back(root->val);
        dfs(root->left, res, depth+1);
        dfs(root->right, res, depth+1);
    }
};
```

* 非递归版本, 用queue实现,用NULL在队里来标注一层的结束

```c++
    vector<vector<int>> levelOrder(TreeNode* root) {
        vector<vector<int>> res;
        if(root == NULL) return res;
        
        vector<int> cur_lev;
        queue<TreeNode*> q;
        q.push(root);
        q.push(NULL); //用NULL表示一层结束
        
        while(!q.empty()){
            TreeNode* cur = q.front();
            q.pop();
            if(cur == NULL){ //一层遍历完了
                res.push_back(cur_lev);
                if(q.size() > 0){
                    q.push(NULL);
                }
                cur_lev.clear();
            }else{
                cur_lev.push_back(cur->val);
                if(cur->left) q.push(cur->left);
                if(cur->right) q.push(cur->right);
            }
        }
        return res;
    }
```
