# [227. Basic Calculator II](https://leetcode.com/problems/basic-calculator-ii/#/description)
* 注意的是用stack时，入栈的时候就把\* 和\\都算好

```C++
    int calculate(string s) {
        int n  = s.length();
        int tmp  =0;
        char sign = '+';
        stack<int> stack;
        
        for(int i =0; i<n; i++){
            if(isdigit(s[i]))
                tmp = tmp*10 + s[i]-'0';
            if(!isdigit(s[i]) && !isspace(s[i]) || i == n -1){
                if(sign== '+')
                    stack.push(tmp);
                else if(sign=='-')
                    stack.push(-tmp);
                else if(sign=='*'){
                    int m = stack.top() * tmp;
                    stack.pop();
                    stack.push(m);
                }else{
                    int m = stack.top() / tmp;
                    stack.pop();
                    stack.push(m);
                }
                
                sign = s[i];
                tmp =0;
            }
        }
        
        int res =0;
        while(!stack.empty()){
            res += stack.top();
            stack.pop();
        }
        
        return res;
```
