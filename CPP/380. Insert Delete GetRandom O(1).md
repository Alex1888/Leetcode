```C++
// 自己写的版本，用map存储数据，只beat了4%
// 而且还是google的怎么在map里取random：std::advance(it, rand()%mymap.size());
class RandomizedSet {
public:
    /** Initialize your data structure here. */
    RandomizedSet() {
        mymap = unordered_map<int, int>();
    }
    
    /** Inserts a value to the set. Returns true if the set did not already contain the specified element. */
    bool insert(int val) {
        if(mymap.find(val) != mymap.end())
            return false;
        mymap[val] = val;
        return true;
    }
    
    /** Removes a value from the set. Returns true if the set contained the specified element. */
    bool remove(int val) {
        if(mymap.find(val) == mymap.end())
            return false;
        mymap.erase(val);
        return true;
    }
    
    /** Get a random element from the set. */
    int getRandom() {
        auto it = mymap.begin();
        std::advance(it, rand()%mymap.size());
        return it->first;
    }
    
public:
    unordered_map<int, int> mymap;
};
```

 ```C++
 // 用vector存储数据，用map存储vector中每个值放入时的次序；
// remove的时候，每次用最后一个值再nums中擦除代替val(目标值)，然后删除最后一个值，巧妙
class RandomizedSet {
public:
    /** Initialize your data structure here. */
    RandomizedSet() {
        
    }
    
    /** Inserts a value to the set. Returns true if the set did not already contain the specified element. */
    bool insert(int val) {
        if(mymap.find(val) != mymap.end())
            return false;
        nums.push_back(val);
        mymap[val] = nums.size()-1;
        return true;
    }
    
    /** Removes a value from the set. Returns true if the set contained the specified element. */
    bool remove(int val) {
        if(mymap.find(val) == mymap.end())
            return false;
        int last = nums.back();
        
        mymap[last] = mymap[val]; //这里要注意次序：是把last和val进行交换，那么就相当于last的index就变成了val的，就相当于说last那个值是在val insert的时候插进来的
        nums[mymap[val]] = last; // 把原来val的值换成last
        nums.pop_back();
        mymap.erase(val);
        return true;
    }
    
    /** Get a random element from the set. */
    int getRandom() {
        return nums[rand() % nums.size()];
    }
    
public:
    unordered_map<int, int> mymap;
    vector<int> nums;
};
 ```
