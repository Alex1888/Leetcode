# [716. Max Stack](https://leetcode.com/problems/max-stack/)
* 非常好的一道题,不是easy难度,至少是medium
* 最开始的想法,很直接的用priority_queue来保存最大值. 但是其实不用,直接用另一个stack保存就行

```c++
class MaxStack {
public:
    /** initialize your data structure here. */
    MaxStack() {
    }
    
    void push(int x) {
        data.push(x);
        cache.push(x);
    }
    
    int pop() {
        int res = data.top();
        data.pop();
        vector<int> tmp;
        while(cache.top() != res){
            tmp.push_back(cache.top());
            cache.pop();
        }
        cache.pop();
        
        int i = tmp.size()-1;
        while(i >= 0){
            cache.push(tmp[i]);
            i--;
        }
        
        return res;
    }
    
    int top() {
        return data.top();
    }
    
    int peekMax() {
        return cache.top();
    }
    
    int popMax() {
        stack<int> tmp;
        while(data.top() != cache.top()){
            tmp.push(data.top());
            data.pop();
        }
        int res = data.top();
        data.pop();
        cache.pop();
        while(!tmp.empty()){
            data.push(tmp.top());
            tmp.pop();
        }
        
        return res;
    }
    
private:
    stack<int> data;
    priority_queue<int> cache;
};
```

*  two stack的O(n)解法

```c++
class MaxStack {
public:
    /** initialize your data structure here. */
    MaxStack() {
        
    }
    
    void push(int x) {
        s.push(x);
        if(maxStack.empty() || x >= maxStack.top()) //注意这里一定要有等于,因为所有最大的数都要在maxStack中
            maxStack.push(x);
    }
    
    int pop() {
        int res = s.top();
        s.pop();
        if(res == maxStack.top())
            maxStack.pop();
        return res;
    }
    
    int top() {
        return s.top();
    }
    
    int peekMax() {
        return maxStack.top();
    }
    
    int popMax() {
        int res = maxStack.top();
        maxStack.pop();
        
        stack<int> tmp;
        while(res != s.top()){
            tmp.push(s.top());
            s.pop();
        }
        
        s.pop();
        while(!tmp.empty()){
            s.push(tmp.top());
            if(maxStack.empty() || tmp.top() >= maxStack.top()) //注意这里一定要有等于
                maxStack.push(tmp.top());
            tmp.pop();
        }
        
        return res;
    }

private:
    stack<int> s;
    stack<int> maxStack;
};


```

* 二刷, 最后的popmax可以直接用定义好的push

```c++
class MaxStack {
public:
    /** initialize your data structure here. */
    MaxStack() {
        
    }
    
    void push(int x) {
        s.push(x);
        if(maxs.empty() || x >= maxs.top()) //每次用maxs之前都要判空
            maxs.push(x);
    }
    
    int pop() {
        int res = s.top();
        s.pop();
        if(!maxs.empty() && res == maxs.top()) //每次用maxs之前都要判空
            maxs.pop();
        return res;
    }
    
    int top() {
        return s.top();
    }
    
    int peekMax() {
        return maxs.top();
    }
    
    int popMax() {
        int res = maxs.top();
        maxs.pop();
        stack<int> t;
        
        while(s.top() != res){
            t.push(s.top());
            s.pop();
        }
        s.pop();
        
        while(!t.empty()){
            push(t.top()); //直接用定义好的push
            t.pop();
        }
        return res;
    }
    
private:
    stack<int> s;
    stack<int> maxs;
};

```
