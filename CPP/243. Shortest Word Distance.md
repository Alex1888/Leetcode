# [243. Shortest Word Distance](https://leetcode.com/problems/shortest-word-distance/#/description)
* 第一反应是用数组存下标

```C++
    int shortestDistance(vector<string>& words, string w1, string w2) {
        unordered_map<string, int> map;
        int n = words.size();

        vector<int> index1, index2;
        for(int i =0; i<n; i++){
            if(words[i] == w1)
                index1.push_back(i);
            else if(words[i] == w2)
                index2.push_back(i);
        }
        
        int res =INT_MAX;
        int l1 = index1.size(), l2 = index2.size();
        for(int i =0; i<l1; i++){
            for(int j =0; j<l2; j++){
                if(abs(index1[i]-index2[j]) < res)
                    res = abs(index1[i]- index2[j]);
            }
        }
        return res;
    }
```

* 但是实际上根本不需要存储啊

```C++
    int shortestDistance(vector<string>& words, string word1, string word2) {
        int n = words.size();
        int p1 = -1, p2 = -1;
        int minpath = INT_MAX;
        for(int i =0; i<n; i++){
            if(words[i] == word1)
                p1 = i;
            if(words[i] == word2)
                p2 = i;
            if(p1!= -1 && p2 != -1)
                minpath = min(minpath, abs(p1-p2));
        }
        
        return minpath;
    }
		```
