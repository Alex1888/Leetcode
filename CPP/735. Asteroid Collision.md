# [735. Asteroid Collision](https://leetcode.com/problems/asteroid-collision/)
* 自己写的版本，注意的是入栈的条件只有stack.top() > 0 && aster < 0
* 如果栈顶是负数，要来的是正数，它俩的方向不同，不可能相遇

```c++
class Solution {
public:
    vector<int> asteroidCollision(vector<int>& asteroids) {
        stack<int> stack;
        for(int aster : asteroids){
        	if(stack.empty()){
        		stack.push(aster);
        		continue;
        	}

            bool needpush = true;
        	while(!stack.empty() &&  (stack.top() > 0 && aster < 0)){
        		if(stack.top() + aster == 0){
        			stack.pop();
        			needpush = false;
                    break;
        		}
    			else if(stack.top() + aster < 0){
    				stack.pop();
    			}else{
                    needpush = false;
                    break;
                }
        		
        	}
        	if(needpush) stack.push(aster);
        }

        int n = stack.size();
        vector<int> res(n, 0);
        int i = n-1;
        while(!stack.empty()){
        	res[i] = stack.top();
        	stack.pop();
            i--;
        }

        return res;
    }
}; 

```

```c++
class Solution {
public:
    vector<int> asteroidCollision(vector<int>& asteroids) {
        vector<int> res; //当做栈来用
        for(int i : asteroids) {
        	while(!res.empty() && res.back() > 0 && res.back() < -i){
        		res.pop_back();
        	}

            //关键在这里：不考虑i<0的情况，因为i如果是特别大的负数，例如-999那么在上面的while就回把栈清空
            // 如果i是小的负数，那只有当栈顶是负数时才入栈；栈顶是正数比这个负数大，就把它消掉了
        	if(res.empty() || i > 0  || res.back() < 0){ 
        		res.push_back(i);
        	}else if(res.back() == -i){
        		res.pop_back();
        	}
        }
        return res;
    }
};
```
