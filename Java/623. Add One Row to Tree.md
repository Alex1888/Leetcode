# [623. Add One Row to Tree](https://leetcode.com/problems/add-one-row-to-tree/description/)

* 自己写的，先level 遍历 得到需要改变的那一层的点

```
class Solution {
    public TreeNode addOneRow(TreeNode root, int val, int depth) {
        if(depth == 1) {
            TreeNode newRoot = new TreeNode(val);
            newRoot.left = root;
            return newRoot;
        }
        
        List<TreeNode> list = new ArrayList();

        levelTrav(root, 1, depth-1, list);
        for(int i=0; i<list.size(); i++) {
            TreeNode cur = list.get(i);
            if(cur != null) {
                TreeNode l = cur.left;
                TreeNode r = cur.right;

                TreeNode newLeft = new TreeNode(val);
                cur.left = newLeft;
                newLeft.left = l;

                TreeNode newRight = new TreeNode(val);
                cur.right = newRight;
                newRight.right = r;
            }
        }

        return root;
    }

    private void levelTrav(TreeNode root, int cur_level, int level, List<TreeNode> list) {
        if(cur_level == level) {
            list.add(root);
            return;
        }

        if(root.left != null) {
            levelTrav(root.left, cur_level+1, level, list);
        }

        if(root.right != null) {
            levelTrav(root.right, cur_level+1, level, list);
        }
    }
}
```
