# [200. Number of Islands](https://leetcode.com/problems/number-of-islands/)
*    http://www.cnblogs.com/ganganloveu/p/4418589.html

```python
class Solution(object):
    def numIslands(self, grid):
        """
        :type grid: List[List[str]]
        :rtype: int
        """
        if len(grid) == 0 : return 0
        if len(grid[0]) == 0: return 0
        n = len(grid); m = len(grid[0])
        re = 0
        for i in range(n):
            for j in range(m):
                if grid[i][j] == '1':
                    self.dfs(grid, i, j, n, m)
                    re += 1
        return re
        
    def dfs(self, grid, i, j, n, m):
        grid[i][j] = 0
        if i>0 and grid[i-1][j] == '1':
            self.dfs(grid, i-1, j, n, m)
            
        if i<n-1 and grid[i+1][j] == '1':
            self.dfs(grid, i+1, j, n, m)
            
        if j>0 and grid[i][j-1] == '1':
            self.dfs(grid, i, j-1, n,m)
            
        if j<m-1 and grid[i][j+1] == '1':
            self.dfs(grid, i, j+1, n,m)                          
```

* DFS的解法, 关键是想到dfs结束后,用grid的值是否是1来判断岛的个数

```c++
class Solution {
public:
    vector<vector<int>> dirs = {{1,0}, {-1, 0}, {0,1}, {0,-1}};
    int numIslands(vector<vector<char>>& grid) {
        if(grid.size() == 0) return 0;
        int n = grid.size(), m = grid[0].size();
        int res = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j] == '1'){
                    res++;
                    dfs(grid, i, j);
                }
            }
        }
        
        return res;
    }
    
    void dfs(vector<vector<char>>& grid, int i , int j){
        if(i < 0 || i >= grid.size() || j <0 || j >= grid[0].size() || grid[i][j] == '0')
            return;
        
        grid[i][j] = '0';        
        for(auto dir : dirs){
            int x = i + dir[0];
            int y = j + dir[1];
            dfs(grid, x, y);
        }
    }
};
```

* BFS的版本, 注意在入队的时候就要把值置为0

```c++
class Solution {
public:
    vector<vector<int>> dirs = {{1,0}, {-1, 0}, {0,1}, {0,-1}};
    int numIslands(vector<vector<char>>& grid) {
        if(grid.size() == 0) return 0;
        int n = grid.size(), m = grid[0].size();
        int res = 0;
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j] == '1'){
                    res++;
                    bfs(grid, i, j);
                }
            }
        }
        
        return res;
    }
    
    void bfs(vector<vector<char>>& grid, int row , int col){
        queue<pair<int,int>> q;
        q.push({row, col});
        
        while(!q.empty()){
            int n = q.size();
            auto cur = q.front(); q.pop();
            int i = cur.first, j = cur.second;
            grid[i][j] = '0';
            for(auto dir : dirs){
                int x = i + dir[0];
                int y = j + dir[1];
                if(x < 0 || x >= grid.size() || y <0 || y >= grid[0].size() || grid[x][y] == '0')
                    continue;
                grid[x][y] = '0'; // 这句很关键:要在入队的时候就把值置为0,不能等到出队的时候再置;因为那可能是很久以后了,而在这期间这个点都是被当成没有被访问来处理的,这样就重复处理了很多次
                q.push({x, y});         
            }
        }
    }
};
```

* 三刷

```java
class Solution {
    int[][] dirs = {{0, 1}, {1, 0}, {0, -1}, {-1, 0}};
    public int numIslands(char[][] grid) {
        // 关键在于怎么数岛的个数，利用每次dfs时把经过的路径都置为0
        // 然后每次出现1就算一个岛
        if(grid.length == 0) return 0;
        int n = grid.length, m = grid[0].length;
        int res = 0;
        
        for(int i=0; i<n; i++){
            for(int j=0; j<m; j++){
                if(grid[i][j] == '1'){
                    res++;
                    dfs(grid, i, j);
                }
            }
        }
        return res;
    }
    
    private void dfs(char[][] grid, int i, int j){
        if(i >= grid.length || i < 0 || j >= grid[0].length || j < 0 || grid[i][j] == '0')
            return;
    
        grid[i][j] = '0';
        for(int[] dir : dirs){
            dfs(grid, i+dir[0], j+dir[1]);
        }
    }
}
```
