# [547. Friend Circles](https://leetcode.com/problems/friend-circles/description/)
* 自己写的并查集标准模板

```c++
class DisjoinSet{
public:
    vector<int> root;
    int cnt; // cnt 表示有几个人是被合并到其他集合里去的了
    DisjoinSet(int n){
        root = vector<int>(n, 0);
        for(int i=0; i<n; i++) root[i] = i;
        cnt = 0;
    }
    
    void Union(int u, int v){
        int a = Find(u);
        int b = Find(v);
        if(a != b){
            root[a] = b;
            cnt++;
        }
    }
    
    int Find(int x){
        return root[x] == x ? x : Find(root[x]);
    }
};


class Solution {
public:
    int findCircleNum(vector<vector<int>>& M) {
        int n = M.size();     
        DisjoinSet djs(n);
        
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(i != j && M[i][j] == 1){
                    djs.Union(i, j);
                }                    
            }
        }
        
        return n - djs.cnt;
    }
};
```

* java 其实是很简单的dfs，不要想复杂了

```java
class Solution {
    int res = 0;
    public int findCircleNum(int[][] M) {
        int m = M.length;
        // 0：unvisited
        for(int i=0; i<m; i++){
            if(M[i][i] == 1){
                res++;
                dfs(M, i);
            }
        }
        
        return res;
    }
    
    private void dfs(int[][] M, int i){
        if(M[i][i] == 0) return;
        
        M[i][i] = 0;
        for(int j=0; j<M.length; j++){
            if(M[i][j] ==1){
                dfs(M, j);
            }
        }
    }
}
```

class Solution {
    public int findCircleNum(int[][] isConnected) {
        int n =isConnected.length;
        UnionFind uf = new UnionFind(n);
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                if(i != j && isConnected[i][j] == 1){
                    uf.union(i, j);
                }
            }
        }
        
        return n - uf.cnt;
    }
}


class UnionFind{
    int[] root;
    int cnt;
    UnionFind(int n){
        root = new int[n];
        cnt = 0;
        for(int i=0; i<n; i++) root[i] = i;
    }
    
    public void union(int u, int v){
        int a = find(u);
        int b = find(v);
        if(a != b){
            root[a] = b;
            cnt++;
        }
    }
    
    public int find(int x){
        return root[x] == x ? x : find(root[x]);
    }
}
