# [23. Merge k Sorted Lists](https://leetcode.com/problems/merge-k-sorted-lists/?tab=Solutions)

```C++
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
 // 我的方法：两两进行insert merge, 比较慢，beat 33%
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int len = lists.size();
        if(len == 0) return NULL;
        if(len == 1) return lists[0];
        ListNode* res = NULL;
        int start = 0;
        // 找到第一个非null
        for(int i = 0; i<len; i++){
            if(lists[i] != NULL){
                res = lists[i];
                start = i;
                break;
            }
        }
        
        // 两两进行insert merge
        for(int i=start+1; i<len; i++){
            if(lists[i] == NULL) continue;
            if(res->val < lists[i]->val)
                res = insertsort(res, lists[i]);
            else
                res = insertsort(lists[i], res);
        }
        
        return res;
    }
    
public:
// l1.value < l2.value
    ListNode* insertsort(ListNode* l1, ListNode* l2){
        ListNode* p1 = l1;
        ListNode* p2 = l2;
        while(p1 != NULL && p2 != NULL){
            // 找到第一个插入点
            while(p1->next != NULL && p1->next->val <= p2->val){
                p1=p1->next;
            }
            
            ListNode* tmp1 = p1->next;
            p1->next = p2;
            
            // 找到l2中插入的末尾节点
            while(p2->next != NULL && tmp1 != NULL && p2->next->val < tmp1->val){
                p2=p2->next;
                p1=p1->next;
            }
                
            ListNode* tmp2 = p2->next;
            p2->next = tmp1;
            p2 = tmp2;
            p1 = p1->next;
        }
        
        return l1;
    }
};

// 这个解法非常简洁明了，可读性相当好，但是实际的时间还不如我写的
ListNode *mergeKLists(vector<ListNode *> &lists) {
    if(lists.empty()){
        return nullptr;
    }
    while(lists.size() > 1){
        lists.push_back(mergeTwoLists(lists[0], lists[1]));
        lists.erase(lists.begin());
        lists.erase(lists.begin());
    }
    return lists.front();
}
ListNode *mergeTwoLists(ListNode *l1, ListNode *l2) {
    if(l1 == nullptr){
        return l2;
    }
    if(l2 == nullptr){
        return l1;
    }
    if(l1->val <= l2->val){
        l1->next = mergeTwoLists(l1->next, l2);
        return l1;
    }
    else{
        l2->next = mergeTwoLists(l1, l2->next);
        return l2;
    }
}

// 分治的解法
class Solution {
public:
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        int n = lists.size();
        if(n == 0) return NULL;
        while(n > 1){
            int k = (n + 1) / 2;
            for(int i=0; i<n / 2; i++){
                lists[i] = mergeList(lists[i], lists[i+k]);
            }
            n = k;
        }
        
        return lists[0];
    }
    
private:
    ListNode* mergeList(ListNode* l1, ListNode* l2) {
        if(l1 == NULL) return l2;
        if(l2 == NULL) return l1;
        
        if(l1->val <= l2->val){
            l1->next = mergeList(l1->next, l2);
            return l1;
        }
        else
        {
            l2->next = mergeList(l1, l2->next);
            return l2;
        }
    }
};
```

* 最优解,速度最快的解法,用优先队列,把排序的动作交给队列.
* 每次取出top的list的第一个元素,肯定是最小的,然后把list->next再放入队列中

```c++
class Solution {
public:
    struct cmp{
        bool operator() (ListNode* l1, ListNode* l2){
            return l1->val > l2->val;
        }
    };
    
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        if(lists.size() == 0) return NULL;
        priority_queue<ListNode*, vector<ListNode*>, cmp> q;
        for(auto list : lists){
            if(list != NULL) q.push(list);
        }
        
        ListNode* head = new ListNode(-1);
        ListNode* cur = head;
        while(!q.empty()){
            ListNode* tmp = q.top(); q.pop();
            if(head == NULL){
                head->next = tmp;
                cur = tmp;
            }else{
                cur->next = tmp;
                cur = cur->next;
            }
            
            if(tmp->next != NULL) q.push(tmp->next);
        }
        
        return head->next;
    }
};
```

```c++
class Solution {
public:
    struct cmp{
        bool operator()(pair<int, ListNode*>& a, pair<int, ListNode*>& b){
            return a.first > b.first;
        }
    };
    ListNode* mergeKLists(vector<ListNode*>& lists) {
        priority_queue<pair<int, ListNode*>, vector<pair<int, ListNode*>>, cmp> q;
        for(auto l : lists) 
            if(l != NULL)q.push({l->val, l});
        
        ListNode* head = new ListNode(-1);
        ListNode* p = head;
        
        while(!q.empty()){
            auto cur = q.top(); q.pop();
            if(cur.second->next != NULL)
                q.push({cur.second->next->val, cur.second->next});
            p->next = new ListNode(cur.first);
            p = p->next;    
        }
        
        return head->next;
    }
};
```

```java
class Solution {
    public ListNode mergeKLists(ListNode[] lists) {
        if(lists == null || lists.length == 0) return null;
        
        PriorityQueue<ListNode> q = new PriorityQueue<>((a, b) -> a.val - b.val);
        
        for(ListNode list : lists){
            if(list != null)
                q.add(list);
        }
        
        ListNode head = new ListNode(0);
        ListNode tail = head;
        
        while(!q.isEmpty()){
            tail.next = q.poll(); // 这里这个顺序要注意
            tail = tail.next; // tail是已经出队的，每次把tail的next加入队列
            
            if(tail.next != null){
                q.add(tail.next);
            }
        }
        
        return head.next;      
    }
}
```
